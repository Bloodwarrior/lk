curse_interaction = {
	category = interaction_category_divine
	icon = curse


	desc = curse_interaction_desc
	
	is_shown = {
		scope:actor = { is_a_divinity = yes }
		NOT = {scope:actor = scope:recipient }
	}

	is_valid_showing_failures_only = {
		scope:recipient = {
			OR = {
				NOT = {
					is_a_divinity = yes
				}
				has_trait = prophet_manifestation
			}
			
		}
	}

	send_options_exclusive = yes
	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = beauty_bad_3 }
			}
		}
		flag = beauty
		localization = beauty_option
	}

	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = intellect_bad_3 }
			}
		}
		flag = intellect
		localization = intellect_option
	}

	send_option = {
		is_valid = {
			scope:recipient =  {
				OR = {
					has_trait = physique_good_1
					has_trait = physique_good_2
					has_trait = physique_good_3
					NOT = { has_trait = strong }
				}
				NOT = { has_trait = physique_bad_3 }
			}
		}
		flag = physique
		localization = physique_option
	}

	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = physique_good_1 }
				NOT = { has_trait = physique_good_2 }
				NOT = { has_trait = physique_good_3 }
				NOT = { has_trait = weak }
			}
		}
		flag = strong
		localization = strong_option
	}

	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = infertile }
			}
		}
		flag = fecund
		localization = fecund_option
	}

	send_option = {
		is_valid = {
		}
		flag = curse_of_defeat
		localization = curse_of_defeat_option
	}

	send_option = {
		is_valid = {
		}
		flag = curse_of_loss
		localization = curse_of_loss_option
	}

	send_option = {
		is_valid = {
		}
		flag = curse_of_deceit
		localization = curse_of_deceit_option
	}

	send_option = {
		is_valid = {
		}
		flag = curse_of_hate
		localization = curse_of_hate_option
	}

	send_option = {
		is_valid = {
		}
		flag = curse_of_sand
		localization = curse_of_sand_option
	}

	send_option = {
		is_valid = {
		}
		flag = curse_of_battle
		localization = curse_of_battle_option
	}

	send_option = {
		is_valid = {
		}
		flag = curse_of_mind
		localization = curse_of_mind_option
	}
 

	on_accept = {
		scope:recipient = {
			add_opinion = {
				target = scope:actor
				modifier = cursed_me_opinion
			}
		}
		if = {
			limit = {
				OR = {
					scope:curse_of_defeat = yes
					scope:curse_of_loss = yes
					scope:curse_of_deceit = yes
					scope:curse_of_hate = yes
					scope:curse_of_sand = yes
					scope:curse_of_battle = yes
					scope:curse_of_mind = yes
				}
			}
			scope:recipient = {
				custom_description = {
					text = increase_investment_text
					object = scope:recipient
					value = 2
					increase_investment_level = { amount = 2 }
				}
			}
		}
		else = {
			scope:recipient = {
				custom_description = {
					text = increase_investment_text
					object = scope:recipient
					value = 1
					increase_investment_level = { amount = 1 }
				}
			}
		}

		if = {
			limit = {
				scope:beauty = yes
			}
			scope:recipient = { 
				decrease_trait_rank = { TRAIT = beauty }
			}
		}

		if = {
			limit = {
				scope:intellect = yes
			}
			scope:recipient = {
				decrease_trait_rank = { TRAIT = intellect }
			}
		}

		if = {
			limit = {
				scope:physique = yes
			}
			scope:recipient = {
				decrease_trait_rank = { TRAIT = physique }
			}
		}

		if = {
			limit = {
				scope:strong = yes
			}
			scope:recipient = {
				if = {
					limit = {
						has_trait = strong
					}
					remove_trait = strong
				}
				else = {
					add_trait = weak
				}
			}
		}

		if = {
			limit = {
				scope:fecund = yes
			}
			scope:recipient = {
				if = {
					limit = {
						has_trait = fecund
					}
					remove_trait = fecund
				}
				else = {
					add_trait = infertile
				}
			}
		}

		scope:recipient = {
			if = {
				limit = {
					scope:curse_of_defeat = yes
				}
				if = {
					limit = {
						has_character_modifier = curse_of_defeat_modifier
					}
					add_character_modifier = lesser_curse_of_defeat_modifier
				}
				else = {
					add_character_modifier = curse_of_defeat_modifier
				}
			}
	
			if = {
				limit = {
					scope:curse_of_loss = yes
				}
				if = {
					limit = {
						has_character_modifier = curse_of_loss_modifier
					}
					add_character_modifier = lesser_curse_of_loss_modifier
				}
				else = {
					add_character_modifier = curse_of_loss_modifier
				}
			}

			if = {
				limit = {
					scope:curse_of_deceit = yes
				}
				if = {
					limit = {
						has_character_modifier = curse_of_deceit_modifier
					}
					add_character_modifier = lesser_curse_of_deceit_modifier
				}
				else = {
					add_character_modifier = curse_of_deceit_modifier
				}
			}

			if = {
				limit = {
					scope:curse_of_hate = yes
				}
				if = {
					limit = {
						has_character_modifier = curse_of_hate_modifier
					}
					add_character_modifier = lesser_curse_of_hate_modifier
				}
				else = {
					add_character_modifier = curse_of_hate_modifier
				}
			}

			if = {
				limit = {
					scope:curse_of_sand = yes
				}
				if = {
					limit = {
						has_character_modifier = curse_of_sand_modifier
					}
					add_character_modifier = lesser_curse_of_sand_modifier
				}
				else = {
					add_character_modifier = curse_of_sand_modifier
				}
			}

			if = {
				limit = {
					scope:curse_of_battle = yes
				}
				if = {
					limit = {
						has_character_modifier = curse_of_battle_modifier
					}
					add_character_modifier = lesser_curse_of_battle_modifier
				}
				else = {
					add_character_modifier = curse_of_battle_modifier
				}
			}

			if = {
				limit = {
					scope:curse_of_mind = yes
				}
				if = {
					limit = {
						has_character_modifier = curse_of_mind_modifier
					}
					add_character_modifier = lesser_curse_of_mind_modifier
				}
				else = {
					add_character_modifier = curse_of_mind_modifier
				}
			}
		}
	
	}

	auto_accept = yes

	cost = {
		piety = {
			add = {
				desc = "BASE_VALUE"
				value = 200
				format = "BASE_VALUE_FORMAT"
			}

			scope:recipient = {
				multiply = {
					desc = "INVESTMENT_COST_MULTIPLY"
					value = investment_level_multiplier
				}
			}

			multiply = {
				desc = "DOMINION_COST_MULTIPLY"
				value = dominion_multiplier
			}
		}
	}
