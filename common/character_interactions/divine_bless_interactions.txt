bless_interaction = {
	category = interaction_category_divine
	icon = bless


	desc = bless_interaction_desc
	
	is_shown = {
		scope:actor = { is_a_divinity = yes }
	}

	is_valid_showing_failures_only = {
		scope:recipient = {
			OR = {
				NOT = {
					is_a_divinity = yes
				}
				has_trait = prophet_manifestation
			}
			
		}
	}

	send_options_exclusive = yes
	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = beauty_good_3 }
			}
		}
		flag = beauty
		localization = beauty_option
	}

	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = intellect_good_3 }
			}
		}
		flag = intellect
		localization = intellect_option
	}

	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = physique_good_3 }
			}
		}
		flag = physique
		localization = physique_option
	}

	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = physique_bad_1 }
				NOT = { has_trait = physique_bad_2 }
				NOT = { has_trait = physique_bad_3 }
				NOT = { has_trait = strong }
			}
		}
		flag = strong
		localization = strong_option
	}

	send_option = {
		is_valid = {
			scope:recipient =  {
				NOT = { has_trait = fecund }
			}
		}
		flag = fecund
		localization = fecund_option
	}

	send_option = {
		is_valid = {
		}
		flag = blessing_of_war
		localization = blessing_of_war_option
	}

	send_option = {
		is_valid = {
		}
		flag = blessing_of_gold
		localization = blessing_of_gold_option
	}

	send_option = {
		is_valid = {
		}
		flag = blessing_of_deceit

		flag = blessing_of_love

		}
		flag = blessing_of_speed
		localization = blessing_of_speed_option

		is_valid = {
		}
		flag = blessing_of_mind
		localization = blessing_of_mind_option
	}

	on_accept = {
		scope:recipient = {
			add_opinion = {
				target = scope:actor
				modifier = blessed_me_opinion
			}
		}
		if = {
			limit = {
				OR = {
					scope:blessing_of_war = yes
					scope:blessing_of_gold = yes
					scope:blessing_of_deceit = yes
					scope:blessing_of_love = yes
					scope:blessing_of_speed = yes
					scope:blessing_of_battle = yes
					scope:blessing_of_mind = yes
				}
			}
			scope:recipient = {
				custom_description = {
					text = increase_investment_text
					object = scope:recipient
					value = 2
					increase_investment_level = { amount = 2 }
				}
			}
		}
		else = {
			scope:recipient = {
				custom_description = {
					text = increase_investment_text
					object = scope:recipient
					value = 1
					increase_investment_level = { amount = 1 }
				}
			}
		}

		if = {
			limit = {
				scope:beauty = yes
			}
			scope:recipient = { 
				increase_trait_rank = { TRAIT = beauty }
			}
		}

		if = {
			limit = {
				scope:intellect = yes
			}
			scope:recipient = {
				increase_trait_rank = { TRAIT = intellect }
			}
		}

		if = {
			limit = {
				scope:physique = yes
			}
			scope:recipient = {
				increase_trait_rank = { TRAIT = physique }
			}
		}

		if = {
			limit = {
				scope:strong = yes
			}
			scope:recipient = {
				if = {
					limit = {
						has_trait = weak
					}
					remove_trait = weak
				}
				else = {
					add_trait = strong
				}
			}
		}

		if = {
			limit = {
				scope:fecund = yes
			}
			scope:recipient = {
				if = {
					limit = {
						has_trait = infertile
					}
					remove_trait = infertile
				}
				else = {
					add_trait = fecund
				}
			}
		}

		scope:recipient = {
			if = {
				limit = {
					scope:blessing_of_war = yes
				}
				if = {
					limit = {
						has_character_modifier = blessing_of_war_modifier
					}
					add_character_modifier = lesser_blessing_of_war_modifier
				}
				else = {
					add_character_modifier = blessing_of_war_modifier
				}
			}
	
			if = {
				limit = {
					scope:blessing_of_gold = yes
				}
				if = {
					limit = {
						has_character_modifier = blessing_of_gold_modifier
					}
					add_character_modifier = lesser_blessing_of_gold_modifier
				}
				else = {
					add_character_modifier = blessing_of_gold_modifier
				}
			}

			if = {
				limit = {
					scope:blessing_of_deceit = yes
				}
				if = {
					limit = {
						has_character_modifier = blessing_of_deceit_modifier
					}
					add_character_modifier = lesser_blessing_of_deceit_modifier
				}
				else = {
					add_character_modifier = blessing_of_deceit_modifier
				}
			}

			if = {
				limit = {
					scope:blessing_of_love = yes
				}
				if = {
					limit = {
						has_character_modifier = blessing_of_love_modifier
					}
					add_character_modifier = lesser_blessing_of_love_modifier
				}
				else = {
					add_character_modifier = blessing_of_love_modifier
				}
			}

			if = {
				limit = {
					scope:blessing_of_speed = yes
				}
				if = {
					limit = {
						has_character_modifier = blessing_of_speed_modifier
					}
					add_character_modifier = lesser_blessing_of_speed_modifier
				}
				else = {
					add_character_modifier = blessing_of_speed_modifier
				}
			}

			if = {
				limit = {
					scope:blessing_of_battle = yes
				}
				if = {
					limit = {
						has_character_modifier = blessing_of_battle_modifier
					}
					add_character_modifier = lesser_blessing_of_battle_modifier
				}
				else = {
					add_character_modifier = blessing_of_battle_modifier
				}
			}

			if = {
				limit = {
					scope:blessing_of_mind = yes
				}
				if = {
					limit = {
						has_character_modifier = blessing_of_mind_modifier
					}
					add_character_modifier = lesser_blessing_of_mind_modifier
				}
				else = {
					add_character_modifier = blessing_of_mind_modifier
				}
			}
		}
	
	}

			scope:recipient = {
				multiply = {
					desc = "INVESTMENT_COST_MULTIPLY"
					value = investment_level_multiplier

			}
			multiply = {
				desc = "DOMINION_COST_MULTIPLY"
				value = dominion_multiplier
			}
		}
